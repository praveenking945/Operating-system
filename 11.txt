#include<stdio.h>
int main()
{
	int p[100],pr[100],wt[100],bt[100],tat[100],pos,temp,i,j,n,avg_wt,avg_tat,total=0;
	printf("enter no of process");
	scanf("%d",&n);
	printf("enter burst time");
	for(i=0;i<=n;i++)
	{
		printf("p%d",i+1);
		scanf("%d",&bt[i]);
		printf("enter priority");
		scanf("%d",&pr[i]);
	}
	p[i]=i+1;
	for(i=0;i<=n;i++)
	{
		pos=i;
		for(j=i+1;j<=n;j++)
		{
			if(pr[j]<pr[pos])
			pos=j;
		}
		temp=pr[i];
		pr[i]=pr[pos];
		pr[pos]=temp;
		temp=bt[i];
		bt[i]=bt[pos];
		bt[pos]=temp;
temp=p[i];
		p[i]=p[pos];
		p[pos]=temp;
	}
	wt[0]=0;
	for(i=1;i<n;i++)
	{
		wt[i]=0;
		for(j=0;j<i;j++)
		wt[i]=wt[i]+bt[i];
		total=total+wt[i];
	}
	avg_wt=total/n;
	for(i=0;i<n;i++)
	{
		tat[i]=wt[i]+bt[i];
		total=total+tat[i];
	}
	avg_tat=total/n;
	printf("average wt=%f",avg_wt);
	printf("average tat=%f",avg_tat);

}
